const URL = "https://vwuekpxqswohphcnqyzl.supabase.co/rest/v1";
const KEY = "eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6InZ3dWVrcHhxc3dvaHBoY25xeXpsIiwicm9sZSI6ImFub24iLCJpYXQiOjE3NzMwOTAzODcsImV4cCI6MjA4ODY2NjM4N30.VhS3kaGBlj58CRs_te30vVcNz5gXiPf0qduUhOAdNx8";
const RESTO_ID = "0908a863-4030-4be3-8a1b-846e211f5b70";

const HEADERS = {
  "apikey": KEY,
  "Authorization": "Bearer " + KEY,
  "Content-Type": "application/json",
};

async function fetchAvis() {
  const res = await fetch(`${URL}/avis?restaurant_id=eq.${RESTO_ID}&order=date_avis.desc`, {
    headers: HEADERS,
  });
  if (!res.ok) throw new Error(await res.text());
  return res.json();
}

async function updateAvis(id, data) {
  const res = await fetch(`${URL}/avis?id=eq.${id}`, {
    method: "PATCH",
    headers: { ...HEADERS, "Prefer": "return=minimal" },
    body: JSON.stringify(data),
  });
  if (!res.ok) throw new Error(await res.text());
}

async function insertClient(data) {
  const res = await fetch(`${URL}/clients`, {
    method: "POST",
    headers: { ...HEADERS, "Prefer": "return=minimal" },
    body: JSON.stringify(data),
  });
  if (!res.ok) throw new Error(await res.text());
}

function getSentiment(note) {
  if (note >= 4) return "positive";
  if (note === 3) return "neutral";
  return "negative";
}

const PLATFORMS = {
  google:      { name: "Google",      color: "#4285F4", icon: "G", bg: "#E8F0FE" },
  facebook:    { name: "Facebook",    color: "#1877F2", icon: "f", bg: "#E7F3FF" },
  yelp:        { name: "Yelp",        color: "#FF1A1A", icon: "Y", bg: "#FFF0F0" },
  tripadvisor: { name: "TripAdvisor", color: "#00AA6C", icon: "T", bg: "#E6F7F2" },
};

function StarRating({ rating, size = 14 }) {
  return (
    <span style={{ display: "inline-flex", gap: 1 }}>
      {[1,2,3,4,5].map(i => (
        <span key={i} style={{ fontSize: size, color: i <= rating ? "#F59E0B" : "#333" }}>★</span>
      ))}
    </span>
  );
}

function PlatformBadge({ platform }) {
  const p = PLATFORMS[platform] || { name: platform, color: "#888", icon: "?", bg: "#222" };
  return (
    <span style={{
      display: "inline-flex", alignItems: "center", gap: 5,
      background: p.bg, color: p.color, border: `1px solid ${p.color}30`,
      borderRadius: 20, padding: "3px 10px", fontSize: 11, fontWeight: 700,
    }}>
      <span style={{ fontWeight: 900, fontSize: 13 }}>{p.icon}</span>
      {p.name}
    </span>
  );
}

function SentimentDot({ sentiment }) {
  const colors = { positive: "#10B981", negative: "#EF4444", neutral: "#F59E0B" };
  return <span style={{ display: "inline-block", width: 8, height: 8, borderRadius: "50%", background: colors[sentiment] || "#888", marginRight: 5 }} />;
}

  const [filterSentiment, setFilterSentiment]   = useState("all");

  const [msgSuccess, setMsgSuccess]       = useState(false);
  const [notification, setNotification]   = useState(null);

        id:        r.id,
        platform:  r.plateforme,
        author:    r.auteur,
        rating:    r.note,
        date:      (r.date_avis || "").split("T")[0],
        text:      r.texte,
        replied:   r.repondu,
        reply:     r.reponse,
        sentiment: getSentiment(r.note),
      })));

      setError(e.message);
    }

          max_tokens: 300,

      setReplyDraft("");

      showNotif("Erreur: " + e.message, "error");

"#EF4444" : "#10B981", color: "#fff", padding: "12px 20px", borderRadius: 12, fontWeight: 600, fontSize: 14, animation: "slideIn 0.3s ease", boxShadow: "0 8px 32px rgba(0,0,0,0.5)" }}>

              <div style={{ display: "grid", gridTemplateColumns: "repeat(4, 1fr)", gap: 16, marginBottom: 28 }}>
                {[
                  { label: "Avis total",      value: stats.total,            icon: "📊", color: "#6366F1" },
                  { label: "Note moyenne",     value: `${stats.avg}/5`,       icon: "⭐", color: "#F59E0B" },
                  { label: "Taux de réponse", value: `${stats.responseRate}%`,icon: "💬", color: "#10B981" },
                  { label: "Sans réponse",     value: stats.pending,          icon: "🔔", color: "#EF4444" },
                ].map(s => (
                  <div key={s.label} style={{ ...card, position: "relative", overflow: "hidden" }}>
                    <div style={{ position: "absolute", top: 16, right: 16, fontSize: 22 }}>{s.icon}</div>
                    <div style={{ color: "#555", fontSize: 11, fontWeight: 600, textTransform: "uppercase", letterSpacing: "0.08em", marginBottom: 8 }}>{s.label}</div>
                    <div style={{ fontSize: 30, fontWeight: 800, fontFamily: "Syne", color: s.color, letterSpacing: "-0.03em" }}>{s.value}</div>
                  </div>
                ))}
              </div>

              {/* Charts */}
              <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 16, marginBottom: 28 }}>
                <div style={card}>
                  <h3 style={{ fontFamily: "Syne", fontWeight: 700, marginBottom: 20, fontSize: 15 }}>Par plateforme</h3>
                  {Object.entries(PLATFORMS).map(([key, p]) => {
                    const count = reviews.filter(r => r.platform === key).length;
                    const avg   = count ? (reviews.filter(r => r.platform === key).reduce((a,r) => a+r.rating,0)/count).toFixed(1) : "—";
                    return (
                      <div key={key} style={{ display: "flex", alignItems: "center", gap: 12, marginBottom: 14 }}>
                        <div style={{ width: 32, height: 32, borderRadius: 8, background: p.bg, display: "flex", alignItems: "center", justifyContent: "center", color: p.color, fontWeight: 900, fontSize: 14, flexShrink: 0 }}>{p.icon}</div>

                          <div style={{ display: "flex", justifyContent: "space-between", marginBottom: 4 }}>
                            <span style={{ fontSize: 13, fontWeight: 600 }}>{p.name}</span>
                            <span style={{ fontSize: 12, color: "#555" }}>{count} avis · {avg}★</span>
                          </div>
                          <div style={{ height: 4, background: "#1A1A24", borderRadius: 2 }}>
                            <div style={{ height: "100%", width: reviews.length ? `${(count/reviews.length)*100}%` : "0%", background: p.color, borderRadius: 2 }} />
                          </div>
                        </div>
                      </div>
                    );
                  })}
                </div>
                <div style={card}>
                  <h3 style={{ fontFamily: "Syne", fontWeight: 700, marginBottom: 20, fontSize: 15 }}>Distribution des notes</h3>
                  {ratingDist.map(({ r, count }) => (

                      <span style={{ fontSize: 13, color: "#F59E0B", width: 20, textAlign: "right" }}>{r}★</span>
                      <div style={{ flex: 1, height: 8, background: "#1A1A24", borderRadius: 4 }}>
                        <div style={{ height: "100%", width: `${(count/maxCount)*100}%`, background: r>=4?"#10B981":r===3?"#F59E0B":"#EF4444", borderRadius: 4 }} />

                        <span style={{ fontWeight: 600, fontSize: 13 }}>{r.author}</span>
                        <StarRating rating={r.rating} size={12} />
                        <span style={{ fontSize: 11, color: "#444" }}>{r.date}</span>
                      </div>
                      <p style={{ fontSize: 13, color: "#777", lineHeight: 1.5 }}>{r.text.substring(0,100)}...</p>
                    </div>
                    <button onClick={() => { setSelectedReview(r); setActiveTab("répondre"); }} style={{ background: "transparent", border: "1px solid #6366F1", color: "#6366F1", borderRadius: 8, padding: "6px 12px", fontSize: 12, fontWeight: 600, cursor: "pointer", whiteSpace: "nowrap" }}>Répondre</button>
                  </div>
                ))}
              </div>
            </div>
          )}

          {/* ── AVIS ── */}
          {activeTab === "avis" && (
            <div style={{ animation: "fadeUp 0.4s ease" }}>
              <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-end", marginBottom: 24, flexWrap: "wrap", gap: 16 }}>
                <div>
                  <h1 style={{ fontFamily: "Syne", fontSize: 28, fontWeight: 800, letterSpacing: "-0.03em" }}>Tous les avis</h1>
                  <p style={{ color: "#555", fontSize: 14, marginTop: 4 }}>{filteredReviews.length} avis affichés</p>
                </div>

                  <div style={{ display: "flex", background: "#111118", border: "1px solid #1E1E2E", borderRadius: 10, padding: 4, gap: 4 }}>
                    {["all",...Object.keys(PLATFORMS)].map(p => (
                      <button key={p} onClick={() => setSelectedPlatform(p)} style={{ padding: "6px 12px", borderRadius: 7, border: "none", cursor: "pointer", background: selectedPlatform===p?"#6366F1":"transparent", color: selectedPlatform===p?"#fff":"#555", fontSize: 12, fontWeight: 600 }}>

                      </button>
                    ))}

                  <div style={{ display: "flex", background: "#111118", border: "1px solid #1E1E2E", borderRadius: 10, padding: 4, gap: 4 }}>

                      <button key={s} onClick={() => setFilterSentiment(s)} style={{ padding: "6px 12px", borderRadius: 7, border: "none", cursor: "pointer", background: filterSentiment===s?"#1E1E2E":"transparent", color: filterSentiment===s?"#E8E8F0":"#555", fontSize: 12, fontWeight: 600 }}>
                        {s==="all"?"Tous":s.charAt(0).toUpperCase()+s.slice(1)}
                      </button>
                    ))}
                  </div>
                </div>
              </div>
              <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>
                {filteredReviews.length === 0 ? (
                  <div style={{ textAlign: "center", padding: 60, color: "#444", ...card }}><div style={{ fontSize: 40, marginBottom: 10 }}>🔍</div><div>Aucun avis trouvé</div></div>
                ) : filteredReviews.map(r => (
                  <div key={r.id} style={{ background: "#111118", border: `1px solid ${r.replied?"#1A2A1A":"#1E1E2E"}`, borderRadius: 14, padding: 20 }}>
                    <div style={{ display: "flex", gap: 14, alignItems: "flex-start" }}>

                    <div style={{ textAlign: "center", padding: 60, color: "#444", ...card }}><div style={{ fontSize: 40, marginBottom: 10 }}>🎉</div><div style={{ fontWeight: 600 }}>Tous les avis ont reçu une réponse!</div></div>
                  ) : pendingReviews.map(r => (
                    <div key={r.id} onClick={() => { setSelectedReview(r); setReplyDraft(""); }} style={{ background: selectedReview?.id===r.id?"#1A1A2E":"#111118", border: `1px solid ${selectedReview?.id===r.id?"#6366F1":"#1E1E2E"}`, borderRadius: 12, padding: 16, cursor: "pointer", transition: "all 0.2s" }}>
                      <div style={{ display: "flex", gap: 10, alignItems: "center", marginBottom: 8 }}>
                        <PlatformBadge platform={r.platform} />
                        <span style={{ fontWeight: 600, fontSize: 13 }}>{r.author}</span>
                        <StarRating rating={r.rating} size={12} />
                        <span style={{ marginLeft: "auto", fontSize: 10, fontWeight: 700, color: r.sentiment==="negative"?"#EF4444":r.sentiment==="positive"?"#10B981":"#F59E0B" }}>
                          <SentimentDot sentiment={r.sentiment} />{r.sentiment}
                        </span>
                      </div>
                      <p style={{ fontSize: 13, color: "#777", lineHeight: 1.5 }}>{r.text.substring(0,120)}...</p>
                      <div style={{ fontSize: 11, color: "#444", marginTop: 8 }}>{r.date}</div>
                    </div>
                  ))}
                </div>
                {selectedReview && (
                  <div style={{ ...card, position: "sticky", top: 80, height: "fit-content" }}>
                    <div style={{ marginBottom: 18, paddingBottom: 16, borderBottom: "1px solid #1E1E2E" }}>
                      <div style={{ display: "flex", gap: 8, alignItems: "center", marginBottom: 8 }}>
                        <PlatformBadge platform={selectedReview.platform} />
                        <span style={{ fontWeight: 700 }}>{selectedReview.author}</span>
                        <StarRating rating={selectedReview.rating} />
                      </div>
                      <p style={{ fontSize: 13, color: "#888", lineHeight: 1.6 }}>{selectedReview.text}</p>
                    </div>
                    <button onClick={() => generateAIReply(selectedReview)} disabled={generating} style={{ width: "100%", padding: "12px", borderRadius: 10, border: "none", background: generating?"#1E1E2E":"linear-gradient(135deg, #6366F1, #8B5CF6)", color: generating?"#555":"#fff", fontWeight: 700, fontSize: 14, cursor: generating?"not-allowed":"pointer", marginBottom: 12, display: "flex", alignItems: "center", justifyContent: "center", gap: 8, fontFamily: "Syne" }}>
                      {generating ? <><span style={{ animation: "pulse 1s infinite" }}>✦</span> Génération...</> : "✦ Générer avec l'IA"}
                    </button>
                    <textarea value={replyDraft} onChange={e => setReplyDraft(e.target.value)} placeholder="Rédigez ou générez une réponse..." style={{ width: "100%", height: 130, background: "#0A0A0F", border: "1px solid #1E1E2E", borderRadius: 10, padding: 14, color: "#E8E8F0", fontSize: 13, lineHeight: 1.6, outline: "none", resize: "none" }} />
                    <div style={{ display: "flex", gap: 10, marginTop: 12 }}>
                      <button onClick={() => { setSelectedReview(null); setReplyDraft(""); }} style={{ flex: 1, padding: "11px", background: "transparent", border: "1px solid #333", borderRadius: 10, color: "#666", fontWeight: 600, cursor: "pointer", fontSize: 13 }}>Annuler</button>
                      <button onClick={() => submitReply(selectedReview.id)} disabled={!replyDraft.trim()} style={{ flex: 2, padding: "11px", background: replyDraft.trim()?"#10B981":"#1A2A1A", border: "none", borderRadius: 10, color: replyDraft.trim()?"#fff":"#444", fontWeight: 700, cursor: replyDraft.trim()?"pointer":"not-allowed", fontSize: 13, fontFamily: "Syne" }}>
                        💾 Sauvegarder ✓
                      </button>
                    </div>
                  </div>
                )}
              </div>
            </div>
          )}

          {/* ── INVITER ── */}
          {activeTab === "inviter" && (
            <div style={{ animation: "fadeUp 0.4s ease", maxWidth: 640 }}>
              <h1 style={{ fontFamily: "Syne", fontSize: 28, fontWeight: 800, marginBottom: 6, letterSpacing: "-0.03em" }}>Inviter des clients</h1>
              <p style={{ color: "#555", marginBottom: 28, fontSize: 14 }}>Les clients sont sauvegardés dans Supabase</p>
              <div style={{ ...card, marginBottom: 20 }}>
                <h3 style={{ fontFamily: "Syne", fontWeight: 700, marginBottom: 20, fontSize: 16 }}>Nouveau client</h3>
                <div style={{ marginBottom: 16 }}>
                  <label style={{ fontSize: 12, color: "#666", fontWeight: 600, textTransform: "uppercase", letterSpacing: "0.08em", display: "block", marginBottom: 8 }}>Nom du client</label>
                  <input value={requestForm.name} onChange={e => setRequestForm(p=>({...p,name:e.target.value}))} placeholder="ex: Marie Tremblay" style={{ width: "100%", padding: "12px 14px", background: "#0A0A0F", border: "1px solid #1E1E2E", borderRadius: 10, color: "#E8E8F0", fontSize: 14, outline: "none" }} />
                </div>
                <div style={{ marginBottom: 16 }}>
                  <label style={{ fontSize: 12, color: "#666", fontWeight: 600, textTransform: "uppercase", letterSpacing: "0.08em", display: "block", marginBottom: 8 }}>Numéro de téléphone</label>
                  <input value={requestForm.phone} onChange={e => setRequestForm(p=>({...p,phone:e.target.value}))} placeholder="+1 (514) 000-0000" style={{ width: "100%", padding: "12px 14px", background: "#0A0A0F", border: "1px solid #1E1E2E", borderRadius: 10, color: "#E8E8F0", fontSize: 14, outline: "none" }} />
                </div>
                <div style={{ marginBottom: 24 }}>
                  <label style={{ fontSize: 12, color: "#666", fontWeight: 600, textTransform: "uppercase", letterSpacing: "0.08em", display: "block", marginBottom: 8 }}>Plateforme cible</label>
                  <div style={{ display: "flex", gap: 8 }}>
                    {Object.entries(PLATFORMS).map(([key,p]) => (
                      <button key={key} onClick={() => setRequestForm(prev=>({...prev,platform:key}))} style={{ flex: 1, padding: "10px 8px", borderRadius: 10, cursor: "pointer", border: `1px solid ${requestForm.platform===key?p.color:"#1E1E2E"}`, background: requestForm.platform===key?`${p.color}20`:"transparent", color: requestForm.platform===key?p.color:"#555", fontSize: 11, fontWeight: 700 }}>
                        <div style={{ fontSize: 16, marginBottom: 2 }}>{p.icon}</div>{p.name}
                      </button>
                    ))}
                  </div>
                </div>
                {requestForm.name && (
                  <div style={{ background: "#0A0A0F", border: "1px solid #1E1E2E", borderRadius: 12, padding: 16, marginBottom: 20 }}>
                    <div style={{ fontSize: 11, color: "#6366F1", fontWeight: 700, marginBottom: 8, textTransform: "uppercase", letterSpacing: "0.08em" }}>📱 Aperçu SMS</div>
                    <p style={{ fontSize: 13, color: "#888", lineHeight: 1.6 }}>Bonjour {requestForm.name} 👋 Merci de votre visite! Pourriez-vous nous laisser un avis sur {PLATFORMS[requestForm.platform].name}? Merci! 🙏</p>
                  </div>
                )}
                <button onClick={sendReviewRequest} disabled={!requestForm.name||!requestForm.phone||sendingMsg||msgSuccess} style={{ width: "100%", padding: "14px", borderRadius: 12, border: "none", background: msgSuccess?"#10B981":(!requestForm.name||!requestForm.phone)?"#1A1A2E":"linear-gradient(135deg, #6366F1, #8B5CF6)", color: (!requestForm.name||!requestForm.phone)?"#444":"#fff", fontWeight: 800, fontSize: 15, cursor: (!requestForm.name||!requestForm.phone)?"not-allowed":"pointer", display: "flex", alignItems: "center", justifyContent: "center", gap: 8, fontFamily: "Syne" }}>
                  {msgSuccess?"✓ Sauvegardé!":sendingMsg?<><span style={{animation:"pulse 1s infinite"}}>●</span> Sauvegarde...</>:"💾 Enregistrer le client"}
                </button>
              </div>
            </div>
          )}

        </div>
      )}
    </div>
  );
}
